# Use an official lightweight Python image
FROM python:3.10-slim

# Set the working directory inside the container
WORKDIR /app

# 1. Install System Dependencies (FFmpeg is critical here)
# We use 'apt-get' because this is a Debian-based Linux image
RUN apt-get update && \
    apt-get install -y ffmpeg && \
    apt-get clean && \
    rm -rf /var/lib/apt/lists/*

# 2. Install Python Dependencies
# Copy requirements first to leverage Docker cache
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 3. Copy the Application Code
COPY . .

# 4. Expose the port (FastAPI default)
EXPOSE 8000

# 5. Start Command
# host 0.0.0.0 is required for Docker containers to be accessible externally
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
